Get a JWT:

curl -v --request POST \
--url http://127.0.0.1:3000/v1/login \
--header "Content-Type: application/json" \
--data '{"username": "admin", "password": "apidocpass"}'

Get and save token to an env var:

TOKEN=$(curl -s --request POST \
  --url http://127.0.0.1:3000/v1/login \
  --header "Content-Type: application/json" \
  --data '{"username": "admin", "password": "apidocpass"}' | jq -r '.token')

Create a patient record:

curl -v --request POST \
--url http://127.0.0.1:3000/v1/patient \
--header "Content-Type: application/json" \
--header "Authorization: Bearer $TOKEN" \
--data '{
  "name": {
    "first": "Bobson",
    "surname": "Dugnutt"
  },
  "address": {
    "address_lines": ["311 Lightning St", "ste 1300"],
    "sublocality": "",
    "locality": "",
    "administrative_area": "",
    "postal_code": "97217",
    "country_region": "USA"
  },
  "birth_date": {
    "day": 8,
    "month": 10,
    "year": 1969
  }
}'

curl -v --request POST \
--url http://127.0.0.1:3000/v1/patient \

Get a patient recrod:

curl -v --request GET \
--url http://127.0.0.1:3000/v1/patient/a6115489-91b0-4ae7-a453-14a5fe2f6f51 \
--header "Content-Type: application/json" \
--header "Authorization: Bearer $TOKEN"

Get a list of patient records:

curl -v --request GET \
--url http://127.0.0.1:3000/v1/patient?first_name="Dingus" \
--header "Content-Type: application/json" \
--header "Authorization: Bearer $TOKEN" \
| json_pp

curl -v --request GET \
--url http://127.0.0.1:3000/v1/patient?birth_year=1983 \
--header "Content-Type: application/json" \
--header "Authorization: Bearer $TOKEN" | json_pp
